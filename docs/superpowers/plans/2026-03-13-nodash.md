# NoDash Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a single `index.html` file (open directly in browser, no server required) that displays ~96 native JavaScript alternatives to Lodash, styled after lodash.com/docs.

**Architecture:** Everything lives in one `index.html` — inline `<style>`, an importmap loading Vue 3 from CDN, and a single `<script type="module">` that defines all components and mounts the app. Child components are defined before parents in script order. No build tools, no npm, no server.

**Tech Stack:** Vue 3 (ESM browser build via unpkg CDN), plain JavaScript, inline CSS, `IntersectionObserver`, `navigator.clipboard`, `window.location.hash`

**Spec:** `docs/superpowers/specs/2026-03-13-nodash-design.md`

Each method object shape (from spec):
```js
{
  id: 'chunk',               // unique, lowercase, no spaces — DOM id and URL hash
  name: '_.chunk',           // lodash method name shown in sidebar
  signature: '_.chunk(array, size)', // shown in h2
  category: 'Array',         // exactly one of: 'Array','Collection','Function','Lang','Object','String'
  description: '...',
  args: [                    // required; [] for zero-arg methods
    { name: 'array', type: 'Array', description: 'The array to process' },
    { name: 'size', type: 'number', description: 'The length of each chunk' },
  ],
  returns: { type: 'Array', description: 'the new array of chunks' }, // null for void
  examples: [
    { label: 'ES6', code: `...` },  // label is 'ES6' or 'ES5'; at least 1 entry
  ],
  compat: [   // exactly 5, always in this order: Chrome, Firefox, Safari, Edge, IE
    { browser: 'Chrome',  version: '45+', supported: true  },
    { browser: 'Firefox', version: '32+', supported: true  },
    { browser: 'Safari',  version: '9+',  supported: true  },
    { browser: 'Edge',    version: '12+', supported: true  },
    { browser: 'IE',      version: null,  supported: false },
  ],
}
```

- [ ] **Step 1: Add `CodeBlock` component definition**

  ```js
  const CodeBlock = defineComponent({
    props: {
      code:  { type: String, required: true },
      label: { type: String, required: true },
    },
    data() {
      return { copyLabel: 'Copy' };
    },
    computed: {
      highlighted() {
        // 1. HTML-escape
        let s = this.code
          .replace(/&/g, '&amp;')
          .replace(/</g, '&lt;')
          .replace(/>/g, '&gt;')
          .replace(/"/g, '&quot;');
        // 2. Comments (must be first to protect comment text from later passes)
        s = s.replace(/(\/\/.*$)/gm, '<span class="cm">$1</span>');
        // 3. Strings (single, double, template)
        s = s.replace(/(`[^`]*`|'[^']*'|"[^"]*")/g, '<span class="str">$1</span>');
        // 4. Numbers
        s = s.replace(/\b(\d+\.?\d*)\b/g, '<span class="num">$1</span>');
        // 5. Keywords
        s = s.replace(
          /\b(const|let|var|return|function|new|if|else|for|of|in|typeof)\b/g,
          '<span class="kw">$1</span>'
        );
        return s;
      },
    },
    methods: {
      copyCode() {
        navigator.clipboard.writeText(this.code).then(() => {
          this.copyLabel = 'Copied!';
          setTimeout(() => { this.copyLabel = 'Copy'; }, 1500);
        }).catch(() => {});
      },
    },
    template: `
      <div>
        <div class="code-label">Example ({{ label }})</div>
        <pre class="code-pre">
          <button class="copy-btn" @click="copyCode">{{ copyLabel }}</button>
          <span v-html="highlighted"></span>
        </pre>
      </div>
    `,
  });
  ```

  ```js
  const Sidebar = defineComponent({
    props: {
      methods:        { type: Array,  required: true },
      activeMethodId: { type: String, default: '' },
    },
    computed: {
      categories() {
        const order = ['Array','Collection','Function','Lang','Object','String'];
        const map = {};
        for (const m of this.methods) {
          if (!map[m.category]) map[m.category] = [];
          map[m.category].push(m);
        }
        return order.filter(cat => map[cat]).map(cat => ({ name: cat, methods: map[cat] }));
      },
    },
    template: `
      <nav class="sidebar">
        <template v-for="cat in categories" :key="cat.name">
          <div class="sidebar-category">{{ cat.name }}</div>
          <a
            v-for="m in cat.methods"
            :key="m.id"
            :href="'#' + m.id"
            :class="['sidebar-method', { active: m.id === activeMethodId }]"
          >{{ m.name }}</a>
        </template>
      </nav>
    `,
  });
  ```

  > Note: sidebar links use `m.name` (e.g. `_.chunk`) not `m.id`. This matches the spec.

### Task 8: MethodList component

**Files:**

- [ ] **Step 1: Add `MethodList` component definition**

  ```js
  const MethodList = defineComponent({
    components: { MethodCard },
    props: {
      methods:     { type: Array,  required: true },
      searchQuery: { type: String, default: '' },
    },
    emits: ['update:activeMethodId'],
    setup(props, { emit }) {
      let observer = null;

      function buildObserver(el) {
        if (observer) observer.disconnect();
        observer = new IntersectionObserver((entries) => {
          const visible = entries.filter(e => e.isIntersecting);
          if (!visible.length) return;

          // Pick topmost: prefer top >= 0, closest to 0; fallback: least negative top
          const positive = visible.filter(e => e.boundingClientRect.top >= 0);
          const chosen = positive.length
            ? positive.reduce((a, b) => a.boundingClientRect.top <= b.boundingClientRect.top ? a : b)
            : visible.reduce((a, b) => a.boundingClientRect.top >= b.boundingClientRect.top ? a : b);

          emit('update:activeMethodId', chosen.target.dataset.methodId);
        }, { threshold: [0, 0.1, 0.3] });

        el.querySelectorAll('section[data-method-id]').forEach(sec => observer.observe(sec));
      }

      const mainRef = ref(null);

      // Rebuild observer whenever filtered methods list changes (after DOM updates)
      watch(() => props.methods, () => {
        nextTick(() => { if (mainRef.value) buildObserver(mainRef.value); });
      }, { flush: 'post' });

      onMounted(() => { if (mainRef.value) buildObserver(mainRef.value); });
      onUnmounted(() => { if (observer) observer.disconnect(); });

      return { mainRef };
    },
    template: `
      <main ref="mainRef">
        <p v-if="methods.length === 0" class="empty-state">
          No methods found for '{{ searchQuery }}'
        </p>
        <MethodCard
          v-for="method in methods"
          :key="method.id"
          :method="method"
        />
      </main>
    `,
  });
  ```

  ```js
  const App = defineComponent({
    components: { TopBar, Sidebar, MethodList },
    setup() {
      const searchQuery   = ref('');
      const activeMethodId = ref('');

      const filteredMethods = computed(() => {
        const q = searchQuery.value.toLowerCase().trim();
        if (!q) return methods;
        return methods.filter(m =>
          m.id.toLowerCase().includes(q) ||
          m.name.toLowerCase().includes(q) ||
          m.category.toLowerCase() === q          // exact match for category
        );
      });

      // Reset activeMethodId when filtered list no longer contains it
      watch(filteredMethods, (list) => {
        if (!list.find(m => m.id === activeMethodId.value)) {
          activeMethodId.value = list[0]?.id ?? '';
        }
      });

      // Hash-based init + hashchange listener for immediate sidebar highlight
      function syncFromHash() {
        const hash = window.location.hash.slice(1);
        if (hash && methods.find(m => m.id === hash)) {
          activeMethodId.value = hash;
        }
      }

      onMounted(() => {
        const hash = window.location.hash.slice(1);
        if (hash && methods.find(m => m.id === hash)) {
          activeMethodId.value = hash;
        } else {
          activeMethodId.value = methods[0]?.id ?? '';
        }
        window.addEventListener('hashchange', syncFromHash);
      });

      onUnmounted(() => {
        window.removeEventListener('hashchange', syncFromHash);
      });

      return { searchQuery, activeMethodId, filteredMethods };
    },
    template: `
      <TopBar v-model="searchQuery" />
      <div class="layout">
        <Sidebar
          :methods="filteredMethods"
          :activeMethodId="activeMethodId"
        />
        <div class="content-wrap">
          <MethodList
            :methods="filteredMethods"
            :searchQuery="searchQuery"
            @update:activeMethodId="activeMethodId = $event"
          />
          <footer>
            <span>Based on <a href="https://github.com/you-dont-need/You-Dont-Need-Lodash-Underscore" target="_blank">You Don't Need Lodash/Underscore</a></span>
            <span>Built with <a href="https://claude.ai/code" target="_blank">Claude Code</a></span>
          </footer>
        </div>
      </div>
    `,
  });

  createApp(App).mount('#app');
  ```
